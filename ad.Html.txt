<script>
  const adsData = JSON.parse(document.getElementById("ads-data").textContent);
  const select = document.getElementById("adSelect");

  function loadAd(index) {
    const ad = adsData[index];
    document.getElementById("product-image").src = ad.image;
    document.getElementById("product-name").textContent = ad.product;
    
    const featuresList = document.getElementById("product-features");
    featuresList.innerHTML = "";
    ad.features.forEach(feature => {
      const li = document.createElement("li");
      li.textContent = feature;
      featuresList.appendChild(li);
    });

    document.getElementById("offer-text").textContent = ad.offer;
    document.getElementById("cta-link").textContent = ad.cta;
    document.getElementById("cta-link").href = ad.link;
  }

  select.addEventListener("change", () => loadAd(select.value));
  loadAd(0); // load first ad by default
